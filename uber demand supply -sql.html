EXTRACT(HOUR FROM request_timestamp) AS request_hour,
COUNT(*) AS total_requests
FROM uber_requests_cleaned
GROUP BY request_hour
ORDER BY request_hour;

--Count of cancellations and no car available
SELECT status, COUNT(*) AS count
FROM uber_requests_cleaned
WHERE status IN ('Cancelled', 'No Cars Available')
GROUP BY status;

--Number of Trip Completions

SELECT
ROUND(
100.0 * SUM(CASE WHEN status = 'Trip Completed' THEN 1 ELSE 0 END) / COUNT(*), 2
) AS completion_percentage
FROM uber_requests_cleaned;

--Average of trip duration
SELECT
ROUND(AVG(DATEDIFF(MINUTE, request_timestamp, drop_timestamp)), 2) AS avg_trip_duration_mins
FROM uber_requests_cleaned
WHERE status = 'Trip Completed';

--Requests of driver not assigned
SELECT COUNT(*) AS unassigned_requests
FROM uber_requests_cleaned
WHERE driver_id IS NULL;

--Trend of requests
SELECT
TO_DATE(request_timestamp) AS request_date,
COUNT(*) AS total_requests
FROM uber_requests_cleaned
GROUP BY request_date
ORDER BY request_date;
